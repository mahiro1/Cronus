			Zeny -= 250;
			warp "alb2trea",43,53;

			if (Zeny < 500) {
				mes "[Marinheiro]";
				mes "Ah! Sem essa! eu preciso dos 500 zeny!";
				close;
			}
			Zeny -= 500;
			if(checkre(0))

			else
				warp "izlude",176,182;
			end;
		case 3:
			mes "[Marinheiro]";
			mes "Faça como quiser....";
			close;
	}
}
